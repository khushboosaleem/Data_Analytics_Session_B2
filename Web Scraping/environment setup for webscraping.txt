# Create a new environment:

conda create --name <your_env_name>

# Active an environment:

conda activate <your_env_name>

# To deactive the current environment:

conda deactivate

# Check all environment:

conda env list

# Check all pkg install in the current environment:

pip list

# remove the environment:

conda remove --name <your_env_name> --all

# Check all environment:

conda <your_env_name> list
